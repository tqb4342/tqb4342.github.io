    3、ITK配准要素

    （1）空间变换

    ①刚性（线性）变换

    刚体变换：位移（水平、垂直）、旋转

    仿射变换：放缩+平移+旋转

    投影变换：变形（但直线投影仍然为直线）

    ②非线性变换：变形（直线变为曲线）

    例如FEM、FFD、B样条形变

    （2）灰度插值（插值器）

    ①最近点插值：距离加权赋值

    ②线性：周围像素的均值

    （3）相似性测度

    ①灰度平均差

    ②正则化相关尺度

    ③梯度微分

    ④互信息度量

    （4）配准参数优化搜索

    ①Powell：极值

    ②梯度下降法

    ③遗传算法

    4、（@todo，这是什么意思？）

    （1）ITK类通过对象工厂实例化：静态类New()，构造和析构是private

    （2）使用itk:ObjectFactoryBase登录一个或多个工厂来控制类的实例

    （3）工厂用CreateInstance（类名）来创建类实例

    （4）工厂主要用于put/output(IO)类，平时New()

    5、Image类基本操作

    （1）首先必须包含图像头文件：

    #include "itkImage.h"

    （2）创建图像指针：

    typedef itk::Image<unsigned short, 3> ImageType;
ImageType::Pointer image = ImageType::New();

    （3）图像区域分类

    LargePossibleRegion：完全图像

    BufferedRegion：内存中的部分图像

    RequestedRegion：滤波要求的图像

    （4）创建图像区域并分配内存：

    ImageType::IndexType start = {{0,0,0}};
ImageType::SizeType size = {{200,200,200}};
ImageType::RegionType region;
region.SetSize(size);
region.SetIndex(start);
image.SetRegions(region);
image.Allocate();

    （5）从文件中读取图像

    #include "itkImageFileReader.h"

typedef unsigned char PixelType;
const unsigned int Dimension = 3;
typedef itk::Image<PixelType , Dimension> ImageType;
//定义Reader指针
typedef itk::ImageFileReader<ImageType> ReaderType;
ReaderType::Pointer reader = ReaderType::New();
//Reader指针更新数据
const char* filename = argv[1];
reader->SetFileName(filename);
reader->Update();
//Image指针获取数据
ImageType::Pointer image = reader->GetOutput();

（6）原点和间距

ImageType::SpacingType spacing = {{0.33,0.33,1.20}};

6、滤波

（1）二值门限处理（分割的最后一步——标记）

#include "itkBinaryThresholdImageFilter.h"


typedef unsigned char InputPixelType;
typedef unsigned char OutputPixelType;


typedef itk::Image<InputPixelType,2> InputImageType;
typedef itk::Image<OutputPixelType,2> OutputImageType;


//二值滤波器
typedef itk::BinaryThresholdImageFilter<InputImageType,OutputImageType> FilterType;


//读图文件指针类型
typedef itk::ImageFileReader<InputImageType> ReaderType;
//写图文件指针类型
typedef itk::ImageFileWriter<OutputImageType> WriterType;


//读图文件指针
ReaderType::Pointer reader = ReaderType::New();
//过滤器管道
FilterType::Pointer filter = FilterType::New();
filter->SetInput(reader->GetOutput());


//设置亮度值
filter->SetOutsideValue(outsideValue);
filter->SetInsideValue(insideValue);


//设置阈值
filter->SetLowerThreshold(lowerThreshold);
filter->SetUpperThreshold(upperThreshold);


//更新
filter->Update();

9、重采样图像滤波器【Filtering/ResampleImageFilter.cxx】

输入：图像、变换、校对机

#include "itkResampleImageFilter.h"
#include "itkAffineTransform.h"
#include "itkNearestNeighborInterpolateImageFunction.h"


const unsigned int Dimension=2;


//像素类型
typedef unsigned char InputPixelType;
typedef unsigned char OutputPixelType;
//图像类型
typedef itk::Image<InputPixelType ,Dimension> InputImageType;
typedef itk::Image<OutputPixelType,Dimension>OutputImageType;
//重采样滤波器
typedef itk::ResampleImageFilter<InputImageType,OutputImageType> FilterType;
FilterType::Pointer filter = FilterType::New();


//变换，这里采用默认变换
typdef itk::AffineTransform<double,Dimension>TransformType;
TransformType::Pointer transform = TransformType::New();
filter->SetTransform(transform);


//最近邻校对机
typdef itk::NearestNeighborInterpolateImageFunction<InputImageType,double>InterpolatorType;
InterpolatorType::Pointer interpolater = InterpolatorType::New();
filter->SetInterpolator(interpolator);
filter->SetDefaultPixelValue(0);//范围外的默认值


//设定原点间距
double spacing[Dimension];
spacing[0] = 1.0;
spacing[1] = 1.0;
filter->SetOutputSpacing(spacing);
double origin[Dimension];
origin[0] = 0.0;
origin[1] = 0.0;
filter->SetOutputOrigin(origin);


//设定重采样范围
InputImageType::SizeType size;
size[0] = 300;
size[1] = 300;
filter->SetSize(size);


filter->SetInput(reader->GetOutput());
writer->SetInput(filter->GetOutput());
writer->Update();
